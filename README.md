<h2>Como usarlo</h2>

<ul>
    <li>composer install</li>
    <li>npm install</li>
    <li>cp .env.example .env</li>
    <li>php artisan key:generate</li>
    <li>php artisan migrate --seed</li>
    <li>php artisan serve</li>
</ul>

<h2>Actualizar</h2>
<ul>
    <li>composer update</li>
    <li>php artisan migrate --seed</li>
</ul>
